public class Main { //Main class, execution will be in this class
	public static void main(String[] args) { //Main method, program beging execution in this method
		Javalin app = Javalin.create(); //Creates an instance  of a Javalin app

		app.start(); //Starts the app

		app.get("foods", ctx -> { //Gets data from the foods program
			Food myFood = new Food("carrots", "vegetable"); //Creates a new food object
			ctx.result("You got: " + myFood.name); //The result of the get
		});
	}
}

/*
This Program creates an application that gets a food object from the foods program and
prints the result of the get to the application
*/
